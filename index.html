// --- CONFIGURAZIONE INSTANTDB --- const APP_ID = 'd1f665e7-bc74-4e27-a8a1-203e49387593'; // Inizializziamo il DB const db = InstantDB.init({ appId: APP_ID }); // --- AUDIO SYSTEM --- const sounds = { soundtrack: new Audio('sountrack.mp3'), frisbee: [new Audio('fx_frisbee1.mp3'), new Audio('fx_frisbee2.mp3'), new Audio('fx_frisbee3.mp3')], merda: [new Audio('fx_merda1.mp3'), new Audio('fx_merda2.mp3'), new Audio('fx_merda3.mp3')], point: new Audio('fx_point.mp3'), schianto: new Audio('fx_schianto.mp3') }; sounds.soundtrack.loop = true; function unlockAudio() { Object.values(sounds).forEach(s => { if (Array.isArray(s)) { s.forEach(t => { t.play().then(() => t.pause()).catch(() => {}); }); } else { s.play().then(() => s.pause()).catch(() => {}); } }); } function playSound(key) { try { if (key === 'soundtrack') { sounds.soundtrack.play().catch(e => console.warn("Autoplay bloccato")); } else if (key === 'frisbee') { const snd = sounds.frisbee[Math.floor(Math.random() * 3)]; snd.currentTime = 0; snd.play(); } else if (key === 'point') { sounds.point.currentTime = 0; sounds.point.play(); } else if (key === 'merda') { const snd = sounds.merda[Math.floor(Math.random() * 3)]; snd.currentTime = 0; snd.play(); } else if (key === 'schianto') { sounds.schianto.currentTime = 0; sounds.schianto.play(); } } catch(e) { console.log("Errore audio:", e); } } // --- CORE GAME --- const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); let score = 0, gameState = 'MENU', poisonedLevel = 0, keys = {}, selectedPlayerName = null; let frameCount = 0, myRole = null; const players = { top: { name: 'Guglielmo', x: 0, y: 50, speed: 7, skinColor: '#ffd393' }, bottom: { name: 'Alessandro', x: 0, y: 0, speed: 7, skinColor: '#ffd393', beardColor: '#d34' } }; const frisbee = { x: 0, y: 0, r: 9, vy: 0, active: false, owner: 'bottom', trail: [], passedThroughBranches: false }; const tree = { x: 0, y: 0, trunkW: 65, trunkH: 85, opening: 130, branchLength: 160, branchWidth: 220, offsetX: 0, drift: 0 }; const musicNpc = { x: 0, y: 0, anim: 0 }; // --- LOGICA DI RETE (SOTTOSCRIZIONE) --- // Qui diciamo a InstantDB di inviarci aggiornamenti per la collezione "room" e "players" db.subscribe({ players: {} }, (result) => { if (result.error) return console.error("Errore DB:", result.error); if (result.data && result.data.players) { result.data.players.forEach(p => { // Se riceviamo dati che non sono i nostri, aggiorniamo la posizione dell'altro if (p.id !== myRole) { if (p.id === 'bottom') { players.bottom.x = p.x; players.bottom.y = p.y; } if (p.id === 'top') { players.top.x = p.x; players.top.y = p.y; } } // Sincronizziamo lo stato globale (punteggio e frisbee) if (p.id === 'gameState') { score = p.score || 0; // Sincronizziamo il frisbee solo se non è attualmente "attivo" per noi if (!frisbee.active) { frisbee.x = p.fx; frisbee.y = p.fy; frisbee.owner = p.owner || 'bottom'; } // Se l'altro ha lanciato, lo attiviamo anche noi if (p.fActive && !frisbee.active) { frisbee.active = true; frisbee.vy = p.fVy; } } }); } }); function syncToDB() { if (!myRole || gameState !== 'PLAYING') return; const me = players[myRole]; // Creiamo una transazione: aggiorniamo noi stessi e lo stato globale const txs = [ db.tx.players[myRole].update({ x: me.x, y: me.y }) ]; // Per evitare conflitti, solo Alessandro (bottom) aggiorna il frisbee ufficialmente nel DB if (myRole === 'bottom') { txs.push(db.tx.players['gameState'].update({ score: score, fx: frisbee.x, fy: frisbee.y, fActive: frisbee.active, fVy: frisbee.vy, owner: frisbee.owner })); } db.transact(txs); } // --- INTERFACCIA --- function goToSelect() { unlockAudio(); playSound('soundtrack'); document.getElementById('screen-title').style.display = 'none'; document.getElementById('screen-select').style.display = 'block'; } function selectChar(name) { selectedPlayerName = name; myRole = (name === 'Alessandro') ? 'bottom' : 'top'; document.getElementById('card-ale').classList.toggle('selected', name === 'Alessandro'); document.getElementById('card-gug').classList.toggle('selected', name === 'Guglielmo'); document.getElementById('startBtn').style.opacity = "1"; document.getElementById('startBtn').style.pointerEvents = "auto"; } document.getElementById('startBtn').addEventListener('click', () => { gameState = 'PLAYING'; document.getElementById('overlay').style.display = 'none'; document.getElementById('ui-ingame').style.display = 'block'; document.getElementById('role-tag').innerText = "PLAYER: " + selectedPlayerName.toUpperCase(); setupPositions(); }); function setupPositions() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; players.top.x = canvas.width / 2; players.bottom.x = canvas.width / 2; players.bottom.y = canvas.height - 110; players.top.y = 80; tree.x = canvas.width / 2; tree.y = canvas.height / 2; musicNpc.x = canvas.width - 100; musicNpc.y = canvas.height / 2; resetFrisbee(); } function resetFrisbee() { frisbee.active = false; frisbee.vy = 0; frisbee.trail = []; frisbee.x = frisbee.owner === 'bottom' ? players.bottom.x : players.top.x; frisbee.y = frisbee.owner === 'bottom' ? players.bottom.y - 45 : players.top.y + 45; } window.addEventListener('keydown', e => { keys[e.code] = true; }); window.addEventListener('keyup', e => { keys[e.code] = false; }); function update() { if (gameState !== 'PLAYING') return; frameCount++; musicNpc.anim = Math.sin(frameCount * 0.15) * 5; poisonedLevel = Math.floor(score / 3); let pTxt = document.getElementById('poison-txt'); if (score >= 3) { pTxt.innerText = "STATUS: TRIPPING (" + (poisonedLevel * 15) + "%)"; pTxt.style.color = "#f0f"; pTxt.classList.add('blink'); tree.drift += 0.01 + (poisonedLevel * 0.01); tree.offsetX = Math.sin(tree.drift) * (140 + score * 5); } const me = players[myRole]; if (keys['ArrowLeft'] && me.x > 30) me.x -= me.speed; if (keys['ArrowRight'] && me.x < canvas.width - 30) me.x += me.speed; if (myRole === 'bottom') { if (keys['ArrowUp'] && me.y > tree.y + 120) me.y -= me.speed; if (keys['ArrowDown'] && me.y < canvas.height - 40) me.y += me.speed; if (keys['Space'] && !frisbee.active && frisbee.owner === 'bottom') { frisbee.active = true; frisbee.vy = -13; playSound('frisbee'); } } else { if (keys['ArrowUp'] && me.y > 40) me.y -= me.speed; if (keys['ArrowDown'] && me.y < tree.y - 130) me.y += me.speed; if (keys['Space'] && !frisbee.active && frisbee.owner === 'top') { frisbee.active = true; frisbee.vy = 13; playSound('frisbee'); } } if (frisbee.active) { frisbee.y += frisbee.vy; frisbee.trail.push({x: frisbee.x, y: frisbee.y}); if (frisbee.trail.length > 12) frisbee.trail.shift(); const relX = frisbee.x - (tree.x + tree.offsetX); if (frisbee.y < tree.y + 20 && frisbee.y > tree.y - tree.branchLength) { if (Math.abs(relX) > tree.opening / 2) { if (Math.abs(relX) < tree.branchWidth / 2) { playSound('schianto'); playSound('merda'); frisbee.active = false; setTimeout(resetFrisbee, 600); } } else { frisbee.passedThroughBranches = true; } } let dTop = Math.hypot(frisbee.x - players.top.x, frisbee.y - players.top.y); let dBot = Math.hypot(frisbee.x - players.bottom.x, frisbee.y - players.bottom.y); if (frisbee.vy < 0 && dTop < 42) { if (frisbee.passedThroughBranches) { score++; playSound('point'); } else { playSound('merda'); } frisbee.owner = 'top'; resetFrisbee(); } else if (frisbee.vy > 0 && dBot < 42) { if (frisbee.passedThroughBranches) { score++; playSound('point'); } else { playSound('merda'); } frisbee.owner = 'bottom'; resetFrisbee(); } if (frisbee.y < -50 || frisbee.y > canvas.height + 50) { playSound('merda'); resetFrisbee(); } } else { frisbee.x = frisbee.owner === 'bottom' ? players.bottom.x : players.top.x; frisbee.y = frisbee.owner === 'bottom' ? players.bottom.y - 40 : players.top.y + 40; } document.getElementById('score-txt').innerText = "SCORE: " + score.toString().padStart(3, '0'); syncToDB(); } function draw() { ctx.fillStyle = '#1e3d1a'; ctx.fillRect(0, 0, canvas.width, canvas.height); if (gameState !== 'PLAYING') { requestAnimationFrame(draw); return; } ctx.save(); if (poisonedLevel > 0 && Math.random() > 0.93) { ctx.translate((Math.random()-0.5) * poisonedLevel * 15, 0); ctx.filter = `hue-rotate(${poisonedLevel * 25}deg) brightness(1.2) contrast(1.5)`; } // Albero ctx.save(); ctx.translate(tree.x + tree.offsetX, tree.y); ctx.fillStyle = '#3d2514'; ctx.fillRect(-tree.trunkW/2, -5, tree.trunkW, tree.trunkH); ctx.fillStyle = '#5d3b22'; ctx.strokeStyle = '#2a1a0e'; ctx.lineWidth = 5; [-1, 1].forEach(side => { ctx.beginPath(); ctx.moveTo(side * (tree.trunkW/2 - 5), 10); ctx.lineTo(side * (tree.branchWidth / 2), -tree.branchLength); ctx.lineTo(side * (tree.opening / 2), -tree.branchLength); ctx.lineTo(side * 5, 30); ctx.closePath(); ctx.fill(); ctx.stroke(); }); ctx.restore(); // NPC Tastierista ctx.save(); ctx.translate(musicNpc.x, musicNpc.y); ctx.fillStyle = '#111'; ctx.beginPath(); ctx.moveTo(-20, 25); ctx.lineTo(20, 25); ctx.lineTo(15, -10); ctx.lineTo(-15, -10); ctx.fill(); ctx.beginPath(); ctx.arc(0, -15, 12, 0, Math.PI*2); ctx.fill(); ctx.fillStyle = '#ffd393'; ctx.beginPath(); ctx.arc(0, -15, 7, 0, Math.PI, false); ctx.fill(); ctx.fillStyle = '#eee'; ctx.fillRect(-25, 15, 50, 10); ctx.restore(); // Frisbee frisbee.trail.forEach((p, i) => { ctx.fillStyle = `rgba(200, 255, 0, ${i/12})`; ctx.beginPath(); ctx.arc(p.x, p.y, frisbee.r * (i/12), 0, Math.PI*2); ctx.fill(); }); ctx.fillStyle = '#dfff00'; ctx.beginPath(); ctx.arc(frisbee.x, frisbee.y, frisbee.r, 0, Math.PI*2); ctx.fill(); // Alessandro ctx.save(); ctx.translate(players.bottom.x, players.bottom.y); ctx.fillStyle = '#444'; ctx.beginPath(); ctx.ellipse(0, 0, 26, 16, 0, 0, Math.PI*2); ctx.fill(); ctx.fillStyle = players.bottom.beardColor; ctx.beginPath(); ctx.arc(0, -6, 13, 0, Math.PI*2); ctx.fill(); ctx.fillStyle = players.bottom.skinColor; ctx.beginPath(); ctx.arc(0, -6, 9, 0, Math.PI*2); ctx.fill(); ctx.restore(); // Guglielmo ctx.save(); ctx.translate(players.top.x, players.top.y); ctx.fillStyle = '#eee'; ctx.beginPath(); ctx.ellipse(0, 0, 26, 16, 0, 0, Math.PI*2); ctx.fill(); ctx.fillStyle = '#431'; ctx.beginPath(); ctx.arc(0, 6, 13, 0, Math.PI*2); ctx.fill(); ctx.fillStyle = players.top.skinColor; ctx.beginPath(); ctx.arc(0, 6, 6, 0, Math.PI*2); ctx.fill(); ctx.restore(); ctx.restore(); for(let i = 0; i < canvas.height; i += 3) { ctx.fillStyle = "rgba(0, 0, 0, 0.15)"; ctx.fillRect(0, i, canvas.width, 1); } update(); requestAnimationFrame(draw); } window.addEventListener('resize', setupPositions); setupPositions(); draw();
